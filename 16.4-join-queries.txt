SELECT ILS.NAME, I.SUMMARY, I.DESCRIPTION FROM ISSUESLISTS ILS JOIN ISSUES I ON ILS.ID = I.ISSUESLIST_ID;

SELECT I.SUMMARY, U.FIRSTNAME, U.LASTNAME FROM ISSUES I JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) FROM USERS U JOIN ISSUES I on U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.LASTNAME, U.FIRSTNAME;
